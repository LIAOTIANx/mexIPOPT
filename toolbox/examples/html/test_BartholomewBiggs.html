function H = hessian(x, sigma, lambda, auxdata)
  x1 = x(1) ;
  x2 = x(2) ;
  x3 = x(3) ;
  x4 = x(4) ;
  Hf = [ 2*x4,      x4,  x4, 2*x1+x2+x3 ; ...
         x4,         0,   0, x1         ; ...
         x4,         0,   0, x1         ; ...
         2*x1+x2+x3, x1, x1, x1         ] ;

  H1 = [ 0, x3*x4, x2*x4, x2*x3 ; ...
         x3*x4, 0, x1*x4, x1*x3 ; ...
         x2*x4, x1*x4, 0, x1*x2 ; ...
         x2*x3, x1*x3, x1*x2, 0 ] ;

     H2 = 2*eye(4,4) ;
  H  = tril(sparse(sigma * Hf + lambda(1)*H1 + lambda(2)*H2)) ;
end

function H = hessianstructure(auxdata)

end
